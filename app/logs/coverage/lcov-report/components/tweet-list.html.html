<dom-module id="tweet-list">
  <template>
    <style>
       :host {
        display: block;
      }
 
    </style>
 
    <dom-repeat items="[[tweets]]" as="tweet">
      <template>
        <tweet-item verbose="[[verbose]]" tweet="[[tweet]]"></tweet-item>
      </template>
    </dom-repeat>
  </template>
 
  <script>
    class TweetList extends Polymer.Element {
      static get is() {
        return 'tweet-list';
      }
 
      static get properties() {
        return {
          /**
           * List of tweets
           */
          tweets: {
            type: Array,
          },
          /**
           * Will pass to tweet-item to render debugging information
           */
          verbose: Boolean,
        };
      }
    }
 
    window.customElements.define(TweetList.is, TweetList);
 
  </script>
</dom-module>
